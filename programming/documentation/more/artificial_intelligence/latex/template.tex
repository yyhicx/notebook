\section{极限}
\flushleft{数列的极限：}
\begin{equation*}
\lim_{n \to \infty} u_n = A
\end{equation*}
\flushleft{函数的极限：}
\begin{gather*}
\lim_{x \to x_0} f(x) = A \\
\lim_{x \to x_0} f(x) = A \text{的充要条件是}\lim_{x \to x_0^-} f(x) = \lim_{x \to x_0^+} f(x) = A
\end{gather*}
